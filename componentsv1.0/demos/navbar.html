<script>
function afterPageLoad() {
    $('#pageContent .menu').menu();
    $('#pageContent .menu .nav li:not(".nav-parent") a').click(function() {
        var $this = $(this);
        $('.menu .nav .active').removeClass('active');
        $this.closest('li').addClass('active');
        var parent = $this.closest('.nav-parent');
        if (parent.length) {
            parent.addClass('active');
        }
    });
    var example1 = $("#js-menu1").superfish();
    var example2 = $("#js-menu2").superfish();
    //example1.children('li:first').superfish('show');

    $('#pageContent').on('click', '.navbar-nav > li > a', function() {
        var $item = $(this).parent('li');
        $item.parent().children('.active').removeClass('active');
        $item.addClass('active');
    });
}
</script>
