artDialog —— 经典的网页对话框组件，内外皆用心雕琢。

详细文档请移步：artDialog官方文档

常用对话框

// alert
$('#js_alert').on('click', function() {
    dialog('提示信息alert', function() {
        alert('您点击确认了！')
    }).showModal();
});

// confirm
$('#js_confirm').on('click', function() {
    dialog('请确认信息！', function() {
        alert('点击确认了')
    }, function() {
        alert('点击取消了')
    }).showModal();
});

// prompt
$('#js_prompt').on('click', function() {
    dialog({
        title: '请输入一个值',
        content: '<input type="text" class="form-control" id="js_prompt_input">',
        ok: function() {
            var value = $('#js_prompt_input').val();
            alert('您输入的值为: ' + value);
        }
    }).showModal();
});
